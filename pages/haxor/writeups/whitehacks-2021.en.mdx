import { useSSG } from 'nextra/ssg'
import {
  getGithubRepoProps,
  GithubRepoCard,
} from '@/components/github-repo-card'

export async function getStaticProps() {
  const repo = await getGithubRepoProps('teamfreestuff/whitehacks-2021')
  return {
    props: {
      ssg: {
        repo: repo,
      },
    },
    revalidate: 300,
  }
}
